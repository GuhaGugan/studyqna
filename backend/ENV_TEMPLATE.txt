# ============================================
# StudyQnA Generator - Environment Configuration
# ============================================
# Copy this file to .env and fill in your values
# Command: cp ENV_TEMPLATE.txt .env
# ============================================

# ============================================
# DATABASE CONFIGURATION
# ============================================
# Format: postgresql://username:password@host:port/database
# Example: postgresql://studyqna_user:mypassword@localhost:5432/studyqna
DATABASE_URL=postgresql://studyqna_user:your_secure_password_here@localhost:5432/studyqna

# ============================================
# JWT SECURITY
# ============================================
# Generate a strong secret key using this command:
# python -c "import secrets; print(secrets.token_urlsafe(32))"
# Copy the output and paste it below
SECRET_KEY=your-super-secret-jwt-key-change-this-in-production
ALGORITHM=HS256
ACCESS_TOKEN_EXPIRE_MINUTES=1440

# ============================================
# ADMIN CONFIGURATION
# ============================================
# Email address that will automatically get admin role when logging in
# This email will be redirected to /admin dashboard
ADMIN_EMAIL=admin@example.com

# ============================================
# STORAGE CONFIGURATION
# ============================================
# Path to storage directory
# For development: ./storage (relative to backend/)
# For production: /home/ubuntu/app/storage (absolute path)
STORAGE_PATH=./storage
ENCRYPT_STORAGE=true

# ============================================
# EMAIL CONFIGURATION
# ============================================
# Choose email provider: "brevo" (recommended) or "smtp"
EMAIL_PROVIDER=brevo

# ============================================
# BREVO API (Recommended - Formerly Sendinblue)
# ============================================
# Get your API key from: https://app.brevo.com/settings/keys/api
# 1. Sign up at https://www.brevo.com (free tier: 300 emails/day)
# 2. Go to Settings > API Keys
# 3. Create a new API key
# 4. Copy the API key and paste it below
# 5. Verify your sender email in Brevo dashboard

BREVO_API_KEY=your-brevo-api-key-here
BREVO_FROM_EMAIL=noreply@yourdomain.com
BREVO_FROM_NAME=StudyQnA

# ============================================
# SMTP CONFIGURATION (Alternative/Backup)
# ============================================
# Only needed if EMAIL_PROVIDER=smtp
# For Gmail Setup:
# 1. Enable 2-Factor Authentication on your Google account
# 2. Go to: https://myaccount.google.com/apppasswords
# 3. Generate an App Password for "Mail"
# 4. Use the generated App Password (16 characters) as SMTP_PASSWORD
#    DO NOT use your regular Gmail password

SMTP_HOST=smtp.gmail.com
SMTP_PORT=587
SMTP_USER=your-email@gmail.com
SMTP_PASSWORD=your-gmail-app-password-16-chars
SMTP_FROM=StudyQnA <noreply@studyqna.com>

# ============================================
# OPENAI CONFIGURATION
# ============================================
# Get your API key from: https://platform.openai.com/api-keys
# Create a new API key and paste it below
# Format: sk-...
OPENAI_API_KEY=sk-your-openai-api-key-here

# AI Usage Threshold (Optional)
# Set monthly token usage threshold for alerts
# Default: 1,000,000 tokens (1M tokens)
# When reached, admin will receive alerts
AI_USAGE_THRESHOLD_TOKENS=1000000

# AI Usage Alert Email (Optional)
# Email address to receive threshold alerts
# Leave empty to disable email alerts (alerts will still show in console)
AI_USAGE_ALERT_EMAIL=admin@yourdomain.com

# ============================================
# APPLICATION CONFIGURATION
# ============================================
APP_NAME=StudyQnA Generator
# Frontend URL (for CORS and redirects)
# Development: http://localhost:3000
# Production: https://your-domain.com
APP_URL=http://localhost:3000

# CORS Origins (comma-separated)
# Development: http://localhost:3000,http://localhost:5173
# Production: https://your-domain.com,https://www.your-domain.com
# Leave empty to use default localhost origins
CORS_ORIGINS=
